227HAL_StatusTypeDef DS3231_SetAlarm1(D3231_Alarm1 *A1_st) {
228 HAL_StatusTypeDef status;
229 uint8_t A1M1 = (A1_st->Mode & 0x01) << 7; // Seconds bit 7.
230 uint8_t A1M2 = (A1_st->Mode & 0x02) << 6; // Minutes bit 7.
231 uint8_t A1M3 = (A1_st->Mode & 0x04) << 5; // Hour bit 7.
232 uint8_t A1M4 = (A1_st->Mode & 0x08) << 4; // Day/Date bit 7.
233 uint8_t DY_DT = (A1_st->Mode & 0x10) << 2; // Day/Date bit 6. Date when 0, day of week when 1.
234
235 uint8_t data[4] = { DS3231_EncodeBCD(A1_st->Seconds) | A1M1,
236 DS3231_EncodeBCD(A1_st->Minutes) | A1M2, DS3231_EncodeBCD(
237 A1_st->Hours) | A1M3, DS3231_EncodeBCD(A1_st->DayDate)
238 | DY_DT | A1M4, };
239
240 status = DS3231_WriteRegisters(DS3231_REG_A1_SECOND, data, 4);
241 if (status != HAL_OK)
242 return status;
243 return DS3231_SetAlarm1IntEn(A1_st->IntEn);
244}

255HAL_StatusTypeDef DS3231_GetAlarm1(D3231_Alarm1 *A1_st) {
256 HAL_StatusTypeDef status;
257 uint8_t data[4];
258 status = DS3231_ReadRegisters(DS3231_REG_A1_SECOND, data, 4);
259 if (status != HAL_OK)
260 return status;
261
262 uint8_t Mode = (data[0] & 0x80) >> 7 // A1M1
263 | (data[1] & 0x80) >> 6 // A1M2
264 | (data[2] & 0x80) >> 5 // A1M3
265 | (data[3] & 0x80) >> 4 // A1M4
266 | (data[3] & 0x40) >> 2; // DY_DT
267 A1_st->Mode = Mode;
268
269 A1_st->Seconds = DS3231_DecodeBCD(data[0] & 0x7F);
270 A1_st->Minutes = DS3231_DecodeBCD(data[1] & 0x7F);
271 A1_st->Hours = DS3231_DecodeBCD(data[2] & 0x3F);
272
273 uint8_t DayDate = (data[3] & 0x40) >> 6;
274 if (DayDate)
275 A1_st->DayDate = DS3231_DecodeBCD(data[3] & 0x0F);
276 else
277 A1_st->DayDate = DS3231_DecodeBCD(data[3] & 0x3F);
278
279 return DS3231_GetAlarm1IntEn(&A1_st->IntEn);
280}

484HAL_StatusTypeDef DS3231_SetDateTime(DS3231_DateTime *dt) {
485 HAL_StatusTypeDef status;
486 uint8_t buffer[7];
487
488 if ((dt->Day >= 1) | (dt->Day <= 7))
489 buffer[3] = DS3231_EncodeBCD(dt->Day);
490 else
491 return HAL_ERROR;
492
493 if ((dt->Date >= 1) | (dt->Date <= 31))
494 buffer[4] = DS3231_EncodeBCD(dt->Date);
495 else
496 return HAL_ERROR;
497
498 if ((dt->Month >= 1) | (dt->Month <= 12))// Century bit implementation is pending
499 buffer[5] = DS3231_EncodeBCD(dt->Month);
500 else
501 return HAL_ERROR;
502
503 if ((dt->Year >= 0) | (dt->Year <= 99))
504 buffer[6] = DS3231_EncodeBCD(dt->Year - 2000U);
505 else
506 return HAL_ERROR;
507
508 if ((dt->Hour_24mode >= 0) | (dt->Hour_24mode <= 23))// Only 24HR mode is supported
509 buffer[2] = DS3231_EncodeBCD(dt->Hour_24mode);
510 else
511 return HAL_ERROR;
512
513 if ((dt->Minute >= 0) | (dt->Minute <= 59))
514 buffer[1] = DS3231_EncodeBCD(dt->Minute);
515 else
516 return HAL_ERROR;
517
518 if ((dt->Second >= 0) | (dt->Second <= 59))
519 buffer[0] = DS3231_EncodeBCD(dt->Second);
520 else
521 return HAL_ERROR;
522
523 status = DS3231_WriteRegisters(DS3231_REG_SECOND, buffer, 7);
524 if (status != HAL_OK)
525 return status;
526
527 uint8_t regCONTROL;
528 status = DS3231_ReadRegister(DS3231_REG_CONTROL, &regCONTROL);
529 if (status != HAL_OK)
530 return status;
531
532 if (dt->Enable == DS3231_ENABLED)
533 regCONTROL &= ~(0x80);
534 else
535 regCONTROL |= (0x80);
536
537 return DS3231_WriteRegister(DS3231_REG_CONTROL, &regCONTROL);
538}

547HAL_StatusTypeDef DS3231_GetDateTime(DS3231_DateTime *dt) {
548 HAL_StatusTypeDef status;
549 uint8_t buffer[7];
550
551 status = DS3231_ReadRegisters(DS3231_REG_SECOND, buffer, 7);
552 if (status != HAL_OK)
553 return status;
554
555 dt->Second = DS3231_DecodeBCD(buffer[0] & 0x7F);
556 dt->Minute = DS3231_DecodeBCD(buffer[1] & 0x7F);
557 dt->Hour_24mode = DS3231_DecodeBCD(buffer[2] & 0x3F);
558 dt->Day = DS3231_DecodeBCD(buffer[3] & 0x07);
559 dt->Date = DS3231_DecodeBCD(buffer[4] & 0x3F);
560 dt->Month = DS3231_DecodeBCD(buffer[5] & 0x1F);
561 dt->Year = DS3231_DecodeBCD(buffer[6]) + 2000U;
562
563 uint8_t regSTATUS;
564 status = DS3231_ReadRegister(DS3231_REG_STATUS, &regSTATUS);
565 if (status != HAL_OK)
566 return status;
567 regSTATUS &= (0x80);
568 regSTATUS >>= DS3231_EOSC;
569
570 dt->Enable = regSTATUS ? DS3231_DISABLED : DS3231_ENABLED;
571
572 return status;
573}

581void DS3231_ToUnixTime(DS3231_DateTime *dt, uint32_t *unixtime) {
582 uint16_t days, years;
583 uint8_t months, hours, minutes, seconds;
584
585 years = dt->Year;
586 months = dt->Month;
587 days = dt->Date;
588 hours = dt->Hour_24mode;
589 minutes = dt->Minute;
590 seconds = dt->Second;
591
592 if (years >= 2000)
593 years -= 2000;
594 else
595 return;
596
597 days -= 1;
598 for (uint8_t i = 1; i < months; i++)
599 days += days_in_month[i - 1];
600
601 if (months > 2 && years % 4 == 0)
602 days++;
603
604 days += (365 * years + (years + 3) / 4);
605
606 *unixtime = ((days * 24UL + hours) * 60 + minutes)
607 * 60+ seconds + SECONDS_FROM_1970_TO_2000;
608}

616void DS3231_ToDateTime(uint32_t *unixtime, DS3231_DateTime *dt) {
617
618 int32_t currYear, daysTillNow, extraTime, extraDays;
619 uint8_t index, day, date, month, flag = 0;
620
621 // Calculate total days unix time T
622 daysTillNow = (*unixtime / (24 * 60 * 60));
623 extraTime = (*unixtime % (24 * 60 * 60));
624 currYear = 1970;
625
626 // Calculating current year
627 while (1) {
628 if (currYear % 400 == 0 || (currYear % 4 == 0 && currYear % 100 != 0)) {
629 if (daysTillNow < 366) {
630 break;
631 }
632 daysTillNow -= 366;
633 } else {
634 if (daysTillNow < 365) {
635 break;
636 }
637 daysTillNow -= 365;
638 }
639 currYear += 1;
640 }
641
642 // Updating extradays because it
643 // will give days till previous day
644 // and we have include current day
645 extraDays = daysTillNow + 1;
646
647 if (currYear % 400 == 0 || (currYear % 4 == 0 && currYear % 100 != 0))
648 flag = 1;
649
650 // Calculating MONTH and DATE
651 month = 0, index = 0;
652 if (flag == 1) {
653 while (1) {
654
655 if (index == 1) {
656 if (extraDays - 29 < 0)
657 break;
658 month += 1;
659 extraDays -= 29;
660 } else {
661 if (extraDays - days_in_month[index] < 0) {
662 break;
663 }
664 month += 1;
665 extraDays -= days_in_month[index];
666 }
667 index += 1;
668 }
669 } else {
670 while (1) {
671 if (extraDays - days_in_month[index] < 0) {
672 break;
673 }
674 month += 1;
675 extraDays -= days_in_month[index];
676 index += 1;
677 }
678 }
679
680 // Current Month
681 if (extraDays > 0) {
682 month += 1;
683 date = extraDays;
684 } else {
685 if (month == 2 && flag == 1)
686 date = 29;
687 else
688 date = days_in_month[month - 1];
689 }
690
691 // Calculating HH:MM:YYYY
692 dt->Date = (uint8_t) date;
693 dt->Month = (uint8_t) month;
694 dt->Year = (uint16_t) currYear;
695 dt->Hour_24mode = extraTime / 3600;
696 dt->Minute = (extraTime % 3600) / 60;
697 dt->Second = (extraTime % 3600) % 60;
698
699 currYear -= month < 3;
700 day = (currYear + currYear / 4 - currYear / 100 + currYear / 400
701 + dow[month - 1] + date) % 7;
702 if (day == 0)
703 day = 7;
704
705 dt->Day = day;
706}
