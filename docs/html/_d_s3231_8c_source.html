225HAL_StatusTypeDef DS3231_SetAlarm1(D3231_Alarm1 *A1_st) {
226 HAL_StatusTypeDef status;
227 uint8_t A1M1 = (A1_st->Mode & 0x01) << 7; // Seconds bit 7.
228 uint8_t A1M2 = (A1_st->Mode & 0x02) << 6; // Minutes bit 7.
229 uint8_t A1M3 = (A1_st->Mode & 0x04) << 5; // Hour bit 7.
230 uint8_t A1M4 = (A1_st->Mode & 0x08) << 4; // Day/Date bit 7.
231 uint8_t DY_DT = (A1_st->Mode & 0x10) << 2; // Day/Date bit 6. Date when 0, day of week when 1.
232 uint8_t data[4] = { DS3231_EncodeBCD(A1_st->Seconds) | A1M1,
233 DS3231_EncodeBCD(A1_st->Minutes) | A1M2,
234 DS3231_EncodeBCD(A1_st->Hours) | A1M3,
235 DS3231_EncodeBCD(A1_st->DayDate)| DY_DT | A1M4 };
236 status = DS3231_WriteRegisters(DS3231_REG_A1_SECOND, data, 4);
237 if (status != HAL_OK)
238 return status;
239 return DS3231_SetAlarm1IntEn(A1_st->IntEn);
240}

251HAL_StatusTypeDef DS3231_GetAlarm1(D3231_Alarm1 *A1_st) {
252 HAL_StatusTypeDef status;
253 uint8_t data[4];
254 status = DS3231_ReadRegisters(DS3231_REG_A1_SECOND, data, 4);
255 if (status != HAL_OK)
256 return status;
257 uint8_t Mode = (data[0] & 0x80) >> 7 // A1M1
258 | (data[1] & 0x80) >> 6 // A1M2
259 | (data[2] & 0x80) >> 5 // A1M3
260 | (data[3] & 0x80) >> 4 // A1M4
261 | (data[3] & 0x40) >> 2; // DY_DT
262 A1_st->Mode = Mode;
263 A1_st->Seconds = DS3231_DecodeBCD(data[0] & 0x7F);
264 A1_st->Minutes = DS3231_DecodeBCD(data[1] & 0x7F);
265 A1_st->Hours = DS3231_DecodeBCD(data[2] & 0x3F);
266 uint8_t DayDate = (data[3] & 0x40) >> 6;
267 if (DayDate)
268 A1_st->DayDate = DS3231_DecodeBCD(data[3] & 0x0F);
269 else
270 A1_st->DayDate = DS3231_DecodeBCD(data[3] & 0x3F);
271 return DS3231_GetAlarm1IntEn(&A1_st->IntEn);
272}

468HAL_StatusTypeDef DS3231_SetDateTime(DS3231_DateTime *dt) {
469 HAL_StatusTypeDef status;
470 uint8_t buffer[7];
471 if ((dt->Day >= 1) | (dt->Day <= 7))
472 buffer[3] = DS3231_EncodeBCD(dt->Day);
473 else
474 return HAL_ERROR;
475 if ((dt->Date >= 1) | (dt->Date <= 31))
476 buffer[4] = DS3231_EncodeBCD(dt->Date);
477 else
478 return HAL_ERROR;
479 if ((dt->Month >= 1) | (dt->Month <= 12))// Century bit implementation is pending
480 buffer[5] = DS3231_EncodeBCD(dt->Month);
481 else
482 return HAL_ERROR;
483 if ((dt->Year >= 0) | (dt->Year <= 99))
484 buffer[6] = DS3231_EncodeBCD(dt->Year - 2000U);
485 else
486 return HAL_ERROR;
487 if ((dt->Hour_24mode >= 0) | (dt->Hour_24mode <= 23))// Only 24HR mode is supported
488 buffer[2] = DS3231_EncodeBCD(dt->Hour_24mode);
489 else
490 return HAL_ERROR;
491 if ((dt->Minute >= 0) | (dt->Minute <= 59))
492 buffer[1] = DS3231_EncodeBCD(dt->Minute);
493 else
494 return HAL_ERROR;
495 if ((dt->Second >= 0) | (dt->Second <= 59))
496 buffer[0] = DS3231_EncodeBCD(dt->Second);
497 else
498 return HAL_ERROR;
499 status = DS3231_WriteRegisters(DS3231_REG_SECOND, buffer, 7);
500 if (status != HAL_OK)
501 return status;
502 uint8_t regCONTROL;
503 status = DS3231_ReadRegister(DS3231_REG_CONTROL, &regCONTROL);
504 if (status != HAL_OK)
505 return status;
506 if (dt->Enable == DS3231_ENABLED)
507 regCONTROL &= ~(0x80);
508 else
509 regCONTROL |= (0x80);
510 return DS3231_WriteRegister(DS3231_REG_CONTROL, &regCONTROL);
511}

520HAL_StatusTypeDef DS3231_GetDateTime(DS3231_DateTime *dt) {
521 HAL_StatusTypeDef status;
522 uint8_t buffer[7];
523 status = DS3231_ReadRegisters(DS3231_REG_SECOND, buffer, 7);
524 if (status != HAL_OK)
525 return status;
526 dt->Second = DS3231_DecodeBCD(buffer[0] & 0x7F);
527 dt->Minute = DS3231_DecodeBCD(buffer[1] & 0x7F);
528 dt->Hour_24mode = DS3231_DecodeBCD(buffer[2] & 0x3F);
529 dt->Day = DS3231_DecodeBCD(buffer[3] & 0x07);
530 dt->Date = DS3231_DecodeBCD(buffer[4] & 0x3F);
531 dt->Month = DS3231_DecodeBCD(buffer[5] & 0x1F);
532 dt->Year = DS3231_DecodeBCD(buffer[6]) + 2000U;
533 uint8_t regSTATUS;
534 status = DS3231_ReadRegister(DS3231_REG_STATUS, &regSTATUS);
535 if (status != HAL_OK)
536 return status;
537 regSTATUS &= (0x80);
538 regSTATUS >>= DS3231_EOSC;
539 dt->Enable = regSTATUS ? DS3231_DISABLED : DS3231_ENABLED;
540 return status;
541}

549void DS3231_ToUnixTime(DS3231_DateTime *dt, uint32_t *unixtime) {
550 uint16_t days, years;
551 uint8_t months, hours, minutes, seconds;
552 years = dt->Year;
553 months = dt->Month;
554 days = dt->Date;
555 hours = dt->Hour_24mode;
556 minutes = dt->Minute;
557 seconds = dt->Second;
558 if (years >= 2000)
559 years -= 2000;
560 else
561 return;
562 days -= 1;
563 for (uint8_t i = 1; i < months; i++)
564 days += days_in_month[i - 1];
565 if (months > 2 && years % 4 == 0)
566 days++;
567 days += (365 * years + (years + 3) / 4);
568 *unixtime = ((days * 24UL + hours) * 60 + minutes)
569 * 60+ seconds + SECONDS_FROM_1970_TO_2000;
570}

578void DS3231_ToDateTime(uint32_t *unixtime, DS3231_DateTime *dt) {
579 int32_t currYear, daysTillNow, extraTime, extraDays;
580 uint8_t index, day, date, month, flag = 0;
581 // Calculate total days unix time T
582 daysTillNow = (*unixtime / (24 * 60 * 60));
583 extraTime = (*unixtime % (24 * 60 * 60));
584 currYear = 1970;
585 // Calculating current year
586 while (1) {
587 if (currYear % 400 == 0 || (currYear % 4 == 0 && currYear % 100 != 0)) {
588 if (daysTillNow < 366) {
589 break;
590 }
591 daysTillNow -= 366;
592 } else {
593 if (daysTillNow < 365) {
594 break;
595 }
596 daysTillNow -= 365;
597 }
598 currYear += 1;
599 }
600 // Updating extradays because it
601 // will give days till previous day
602 // and we have include current day
603 extraDays = daysTillNow + 1;
604 if (currYear % 400 == 0 || (currYear % 4 == 0 && currYear % 100 != 0))
605 flag = 1;
606 // Calculating MONTH and DATE
607 month = 0, index = 0;
608 if (flag == 1) {
609 while (1) {
610 if (index == 1) {
611 if (extraDays - 29 < 0)
612 break;
613 month += 1;
614 extraDays -= 29;
615 } else {
616 if (extraDays - days_in_month[index] < 0) {
617 break;
618 }
619 month += 1;
620 extraDays -= days_in_month[index];
621 }
622 index += 1;
623 }
624 } else {
625 while (1) {
626 if (extraDays - days_in_month[index] < 0) {
627 break;
628 }
629 month += 1;
630 extraDays -= days_in_month[index];
631 index += 1;
632 }
633 }
634 // Current Month
635 if (extraDays > 0) {
636 month += 1;
637 date = extraDays;
638 } else {
639 if (month == 2 && flag == 1)
640 date = 29;
641 else
642 date = days_in_month[month - 1];
643 }
644 // Calculating HH:MM:YYYY
645 dt->Date = (uint8_t) date;
646 dt->Month = (uint8_t) month;
647 dt->Year = (uint16_t) currYear;
648 dt->Hour_24mode = extraTime / 3600;
649 dt->Minute = (extraTime % 3600) / 60;
650 dt->Second = (extraTime % 3600) % 60;
651 currYear -= month < 3;
652 day = (currYear + currYear / 4 - currYear / 100 + currYear / 400
653 + dow[month - 1] + date) % 7;
654 if (day == 0)
655 day = 7;
656 dt->Day = day;
657}
